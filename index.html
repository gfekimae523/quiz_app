// app.js const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const path = require('path'); const app = express(); const server = http.createServer(app); const io = new Server(server); let firstResponder = null; let participants = {}; app.use(express.static(path.join(__dirname, 'public'))); io.on('connection', (socket) => { console.log('A user connected:', socket.id); // Handle nickname submission socket.on('set_nickname', (nickname) => { participants[socket.id] = nickname; console.log(`Nickname set: ${nickname}`); }); // Handle button press socket.on('buzz', () => { if (!firstResponder) { firstResponder = participants[socket.id] || 'Unknown'; io.emit('lock_buzzers', firstResponder); console.log(`First responder: ${firstResponder}`); } }); // Handle reset from admin socket.on('reset', () => { firstResponder = null; io.emit('reset_buzzers'); console.log('Buzzers reset'); }); // Disconnect socket.on('disconnect', () => { console.log('A user disconnected:', socket.id); delete participants[socket.id]; }); }); server.listen(3000, () => { console.log('Server is running on http://localhost:3000'); }); // public/index.html Quiz App
